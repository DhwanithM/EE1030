\begin{align}
	d&=\norm{\vec P-\vec O}\\
 \implies d&=\sqrt{\brak{\vec P-\vec O}^\top\brak{\vec P-\vec O}}\\
 \implies d&=\sqrt{\myvec{-6\\8}\myvec{-6&8}}\\
 \implies d&=\sqrt{\brak{-6}^2+8^2}\\
 \implies d&=\sqrt{100}\\
 \implies d&=10
    \end{align}
